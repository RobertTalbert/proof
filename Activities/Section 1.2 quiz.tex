\begin{questions}
	
	\question[1] The first line of a direct proof of the statement ``If $A$, then $B$'' is
	\begin{parts}
		\part Assume that $A$ is true
		\part Assume that $B$ is true
		\part Assume that $A$ is false
		\part Assume that $B$ is false
		\part Assume that the entire statement ``If $A$, then $B$'' is true
	\end{parts}
	
	
	\question[1] Let's say that an integer $a$ is ``type 0'' if there exists an integer $n$ such that $a = 3n$. Which of the following are type 0 integers? 
	\begin{parts}
		\part $18$
		\part $-11$
		\part $0$
		\part $3\pi$
		\part All of the above
		\part All of (a), (b), and (c)
		\part Just (a) and (c)
	\end{parts}
	
	\question[1] Which of the following are among the guidelines for writing proofs as discussed in the text? 
	\begin{parts}
		\part Use the pronoun ``we''. 
		\part Begin the proof with a statement of your assumptions. 
		\part Before starting the proof, begin with a carefully worded statement of the theorem or result to be proven. 
		\part Display important equations and mathematical expressions. 
		\part All of the above
	\end{parts}
